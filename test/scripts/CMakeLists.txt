macro(pytest testname)
  add_test(${testname}
    /usr/bin/env
    PYTHONPATH=${LIBRARY_OUTPUT_PATH}:${CMAKE_SOURCE_DIR}/python:$PYTHONPATH
    ${ARGN}
    )
endmacro()
